In this code example we'll look at a C# containing class and a .razor page being used to create a CNC calculator 


                    //This is the library we will use for are validation
                    using System.ComponentModel.DataAnnotations;
 
                    //This is the namespace we will be using for this class
                    namespace Common.Services
                    {
                        //This is the class
                        public class CNCFeedRateCalc
                        {
                            //Default constructor with default values
                            public CNCFeedRateCalc() : this(0.0f, 2, 0.1f, "mm")
                            {
                            }
                            //This is the constructor that takes inputs
                            public CNCFeedRateCalc(float rpm, int numFlutes, float chipLoad, string units)
                            {
                                this.rpm = rpm;
                                this.numFlutes = numFlutes;
                                this.chipLoad = chipLoad;
                                this.Units = units;
                                CalcFeeRate();
                            }
                            //rpm variable
                            private float rpm = 0.0f;
                            //This sets the requirements that the variable needs to reach
                            [Required]
                            [Range(0, 100000)]
                            //getters and setters for the variabke
                            public float RPM
                            {
                                get { return rpm; }
                                //when setting the value for rpm also calls the calculate function
                                set { rpm = value; CalcFeeRate(); }
                            }
                    
                            private int numFlutes;
                    
                            //Validation for the numFlutes variable
                            [Required]
                            [Range(1, int.MaxValue, ErrorMessage = "Must have at least one flute.")]
                            public int NumFlutes
                            {
                                get { return numFlutes; }
                                //Again when settings the varaible call the calculate function
                                set { numFlutes = value; CalcFeeRate(); }
                            }
                    
                            private float chipLoad;
                            
                            //Validation for the chipLoad variable
                            [Required]
                            [Range(0, float.MaxValue)]
                            public float ChipLoad
                            {
                                get { return chipLoad; }
                                //Again make sure to call the calculate function when setting
                                set { chipLoad = value; CalcFeeRate(); }
                            }
                    
                            //Set required variables again
                            [Required]
                            public string Units { get; set; } = "mm";
                            public float FeedRate { get; private set; }
                    
                            //This is the calculate function that will calculate the FeedRate for the page
                            private void CalcFeeRate()
                            {
                                FeedRate = rpm * numFlutes * chipLoad;
                            }
                        }
                    }

Database persistence in blazor

To gain object persistence in blazor you need the right packages these are the packages that you'll need 


                    Microsoft.EntityFrameworkCore
                    Microsoft.EntityFrameworkCore.SqlServer  (or .Sqlite if using SQLite)
                    Microsoft.EntityFrameworkCore.Tools

                    //Running these commands in the terminal will get you the packages
                    dotnet add package Microsoft.EntityFrameworkCore.SqlServer
                    dotnet add package Microsoft.EntityFrameworkCore.Tools

After hooking up the DBContext you need to create a repository which needs a class and a interface 


                    using Common.Models;

                    public interface IDataService
                    {
                        Task> GetBalanceSheetItemsAsync();
                        Task AddBalanceSheetLineItemAsync(BalanceSheetLineItem item);
                        Task DeleteBalanceSheetLineItemAsync(int id);
                    }
                

                    using Microsoft.EntityFrameworkCore;
                    using Common.Data; // Your DbContext namespace
                    using Common.Models;

                    public class DatabaseDataService : IDataService
                    {
                        private readonly ApplicationDbContext _context;

                        public DatabaseDataService(ApplicationDbContext context)
                        {
                            _context = context;
                        }

                        public async Task> GetBalanceSheetItemsAsync()
                        {
                            return await _context.BalanceSheetLineItems.ToListAsync();
                        }

                        public async Task AddBalanceSheetLineItemAsync(BalanceSheetLineItem item)
                        {
                            _context.BalanceSheetLineItems.Add(item);
                            await _context.SaveChangesAsync(); 
                        }

                        public async Task DeleteBalanceSheetLineItemAsync(int id)
                        {
                            var item = await _context.BalanceSheetLineItems.FindAsync(id);
                            if (item != null)
                            {
                                _context.BalanceSheetLineItems.Remove(item);
                                await _context.SaveChangesAsync();
                            }
                        }

                        public async Task AddContactSubmissionAsync(ContactSubmission submission)
                        {
                            _context.ContactSubmissions.Add(submission);
                            await _context.SaveChangesAsync();
                        }

                        public async Task> GetContactSubmissionsAsync()
                        {
                            return await _context.ContactSubmissions.OrderByDescending(c => c.SubmittedAt).ToListAsync();
                        }
                    }

                        // ── Master state (Database-backed) ────────────────────────────────────────

                        private List allItems = new();
                        
                        // These "computed" lists filter the master list into the two columns
                        private List assets => allItems.Where(x => x.IsAsset).ToList();
                        private List liabilities => allItems.Where(x => !x.IsAsset).ToList();

                        // ── Dialog & UI state ─────────────────────────────────────────────────────

                        private bool showDialog = false;
                        private bool dialogIsAsset = true;
                        private string dialogSection = "Asset";
                        private BalanceSheetLineItem? editingItem = null;

                        private string? uploadedFileName;
                        private DateTime? uploadedAt;

                        // ── Computed Totals

                        private decimal TotalAssets => BalanceSheetService.GetTotal(assets);
                        private decimal TotalLiabilities => BalanceSheetService.GetTotal(liabilities);
                        private decimal Equity => BalanceSheetService.GetEquity(assets, liabilities);

                        // ── Lifecycle & Database Refresh ──────────────────────────────────────────

                        protected override async Task OnInitializedAsync()
                        {
                            await RefreshFromDatabase();
                        }

                        private async Task RefreshFromDatabase()
                        {
                            allItems = await DataService.GetBalanceSheetItemsAsync();
                        }

                        // ── Event Handlers ────────────────────────────────────────────────────────

                        private void OpenDialog(bool isAsset)
                        {
                            dialogIsAsset = isAsset;
                            dialogSection = isAsset ? "Asset" : "Liability";
                            editingItem = new BalanceSheetLineItem { IsAsset = isAsset }; 
                            showDialog = true;
                        }

                        private void OpenEditDialog(BalanceSheetLineItem item, bool isAsset)
                        {
                            dialogIsAsset = isAsset;
                            dialogSection = isAsset ? "Asset" : "Liability";
                            editingItem = item; 
                            showDialog = true;
                        }

                        private async Task HandleDialogSave(BalanceSheetLineItem result)
                        {
                            result.IsAsset = dialogIsAsset;

                            // If Id is 0, it's a brand new entry for the DB
                            if (result.Id == 0)
                            {
                                await DataService.AddBalanceSheetLineItemAsync(result);
                            }
                            else 
                            {

                            }

                            await RefreshFromDatabase();
                            CloseDialog();
                        }

                        private async Task RemoveItem(BalanceSheetLineItem item, bool isAsset)
                        {
                            await DataService.DeleteBalanceSheetLineItemAsync(item.Id);
                            await RefreshFromDatabase();
                        }

                        private void CloseDialog()
                        {
                            showDialog = false;
                            editingItem = null;
                        }

                        private void HandleDialogCancel() => CloseDialog();

                        // ── Export / Upload ──────────────────────────────────────────────────────

                        private async Task ExportDocx()

                            var bytes = DocxBuilder.Build(assets, liabilities, TotalAssets, TotalLiabilities, Equity);

                            await JS.InvokeVoidAsync("downloadFile", "balance-sheet.docx",
                                "application/vnd.openxmlformats-officedocument.wordprocessingml.document", base64);
                        }

                        private void HandleDocxUpload(InputFileChangeEventArgs e)

                            uploadedFileName = e.File.Name;
                            uploadedAt = DateTime.Now;
                        }
